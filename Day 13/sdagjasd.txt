app.post('/chandra/:id',(req,res) => {
    const newUser = req.body;
    chandra.push(newUser);
    res.json(newUser)
})

app.delete('/chandra/:id',(req,res) => {
   const idToDelete = Number(req.params.id);

    const index = chandra.findIndex(obj => obj.id === idToDelete);

    if (index === -1) {
        return res.status(404).send("User not found");
    }

    const deletedUser = chandra.splice(index, 1);

    res.json(deletedUser)
})
